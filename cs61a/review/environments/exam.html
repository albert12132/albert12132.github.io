Question 1

def funny(joke):
    hoax = joke + 1
    return funny(hoax)

def sad(joke):
    hoax = joke - 1
    return hoax + hoax

funny, sad = sad, funny
result = funny(sad(1))

    return double(x + x)

Question 3

    def time(time):
        return fun(x)
    x = 4
    return time

    return x**2
    x = 5

result = fun(boo)(10)
