Q1

def factorial(n):
    if n == 0:
        return 1
    return n * factorial(n-1)

ans = factorial(2)

Q2

def foo(n):
    i = 0
    if n == 0:
        return 0
    result = foo(n - 2)
    i += 1
    return i + result

result = foo(4)

Q3

def bar(f):
    def g(x):
        if x == 1:
            return f(x)
        else:
            return f(x) + g(x - 1)
    return g

f = 4
bar(lambda x: x + f)(2)
